grade12"><rb>柴</rb><rp>(</rp><rt>시</rt><rp>)</rp></ruby><ruby class="hanja api grade10"><rb>棍</rb><rp>(</rp><rt>곤</rt><rp>)</rp></ruby></b>으로 불렸으며, <ruby class="hanja api grade42"><rb>分斷</rb><rp>(</rp><rt>분단</rt><rp>)</rp></ruby> <ruby class="hanja api grade52"><rb>當時</rb><rp>(</rp><rt>당시</rt><rp>)</rp></ruby>에는 <ruby class="hanja api grade32"><rb>越南</rb><rp>(</rp><rt>월남</rt><rp>)</rp></ruby> <ruby class="hanja api grade62"><rb>南部</rb><rp>(</rp><rt>남부</rt><rp>)</rp></ruby>의 <ruby class="hanja api </p><p><ruby class="hanja api grade32"><rb>越南</rb><rp>(</rp><rt>월남</rt><rp>)</rp></ruby> <ruby class="hanja api grade62"><rb>南東部</rb><rp>(</rp><rt>남동부</rt><rp>)</rp></ruby>에 <ruby class="hanja api grade42"><rb>位置</rb><rp>(</rp><rt>위치</rt><rp>)</rp></ruby>한 이곳은 <ruby class="hanja api grade32"><rb>越南</rb><rp>(</rp><rt>월남</rt><rp>)</rp></ruby>의 <ruby class="hanja api grade42"><rb>經濟</rb><rp>(</rp><rt>경제</rt><rp>)</rp></ruby>, <ruby class="hanja api grade32"><rb>貿易</rb><rp>(</rp><rt>무역</rt><rp>)</rp></ruby>, <ruby class="hanja api grade60"><rb>交通</rb><rp>(</rp><rt>교통</rt><rp>)</rp></ruby>, <ruby class="hanja api grade52"><rb>文化</rb><rp>(</rp><rt>문화</rt><rp>)</rp></ruby>의 <ruby class="hanja api grade70"><rb>中心地</rb><rp>(</rp><rt>중심지</rt><rp>)</rp></ruby>이다. 이 <ruby class="hanja api grade50"><rb>都市</rb><rp>(</rp><rt>도시</rt><rp>)</rp></ruby>의 <ruby class="hanja api grade60"><rb>現在</rb><rp>(</rp><rt>현재</rt><rp>)</rp></ruby> 이름은 <ruby class="hanja api grade62"><rb>現代</rb><rp>(</rp><rt>현대</rt><rp>)</rp></ruby> <ruby class="hanja api grade32"><rb>越南</rb><rp>(</rp><rt>월남</rt><rp>)</rp></ruby>의 <ruby class="hanja api grade80"><rb>國父</rb><rp>(</rp><rt>국부</rt><rp>)</rp></ruby>이자 <ruby class="hanja api grade62"><rb>共</rb><rp>(</rp><rt>공</rt><rp>)</rp></ruby><ruby class="hanja api grade0"><rb>產</rb><rp>(</rp><rt>산</rt><rp>)</rp></ruby><ruby class="hanja api grade42"><rb>黨</rb><rp>(</rp><rt>당</rt><rp>)</rp></ruby> <ruby class="hanja api grade42"><rb>黨</rb><rp>(</rp><rt>당</rt><rp>)</rp></ruby><ruby class="hanja api grade60"><rb>主席</rb><rp>(</rp><rt>주석</rt><rp>)</rp></ruby>인 <a href="/wiki/%E8%83%A1%E5%BF%97%E6%98%8E" title="胡志明"><ruby class="hanja api grade32"><rb>胡志明</rb><rp>(</rp><rt>호지명</rt><rp>)</rp></ruby></a>을 기리기 <ruby class="hanja api grade42"><rb>爲</rb><rp>(</rp><rt>위</rt><rp>)</rp></ruby>해 지어졌다.
